@Override
	public FiltersDTO getActiveRecordsByAgeFilteredData(FiltersDTO activieRecordsByAgeFilteredData) {
		FiltersDTO filteredData = null;
		try {
			filteredData = authServiceCaller.getWebClient().post()
					.uri("http://localhost:8007/polestar/unit/api/filters/v1.0/getProfileFilter").body(BodyInserters.fromValue(activieRecordsByAgeFilteredData))
					.retrieve().bodyToMono(FiltersDTO.class).block();
		} catch (Exception e) {
			LOGGER.error("error in retrieving userrole by userId", e);
		}
		return filteredData;
	}

@Override
	public List<activityDTO> getListOfUnitIdFromProfileToActivieRecordsByAge(FiltersDTO profilefilteredData,
			List<activityDTO> activityList) {
		List<activityDTO> filteredList = new ArrayList<>();
		List<activityDTO> appendfilteredList = new ArrayList<>();
		List<String> listOfOrgs = new ArrayList<>();
		List<String> listOfSubOrgs = new ArrayList<>();
		List<String> listOfStatus = new ArrayList<>();
		List<String> listOfRequestType = new ArrayList<>();
		List<String> listOfAssignedTo = new ArrayList<>();
		List<String> listOfAging = new ArrayList<>();
		
		
		if(profilefilteredData.getOrgs()!=null) {
			List<Options> orgs = profilefilteredData.getOrgs();
			for (Options options : orgs) {
				listOfOrgs.add(options.getId());
			}
			activityList.forEach(activityDto->{
				Optional<activityDTO> filteredActivityDto = listOfOrgs.stream()
				 .filter(filteredOrgs->activityDto.getTenantName().equals(filteredOrgs))
				 .map(filteredOrgs->activityDto)
				 .findFirst();
			filteredActivityDto.ifPresent(filteredList::add);
			});
		}
		if(profilefilteredData.getSubOrgs()!=null) {
			List<Options> subOrgs = profilefilteredData.getSubOrgs();
			for (Options options : subOrgs) {
				listOfSubOrgs.add(options.getId());
			}
			if(!filteredList.isEmpty()) {
				appendfilteredList.addAll(filteredList);
			}else {
				appendfilteredList.addAll(activityList);
			}
			filteredList.clear();
			for(activityDTO activityDto : appendfilteredList) {
				for(String filteredSubOrgs : listOfSubOrgs) {
					if(activityDto.getUnitName().equalsIgnoreCase(filteredSubOrgs)) {
						filteredList.add(activityDto);
					}
			}
			}
			appendfilteredList.clear();
		}
		if(profilefilteredData.getStatus()!=null) {
			listOfStatus = profilefilteredData.getStatus();
			if(!filteredList.isEmpty()) {
				appendfilteredList.addAll(filteredList);
			}else {
				appendfilteredList.addAll(activityList);
			}
			filteredList.clear();
			for(activityDTO activityDto : appendfilteredList) {
				for(String status : listOfStatus) {
					if(activityDto.getStatus().equalsIgnoreCase(status) && !(filteredList.contains(activityDto))) {
						filteredList.add(activityDto);
					}
			}
			}
			appendfilteredList.clear();
		}
		if(profilefilteredData.getRequestType()!=null) {
			listOfRequestType = profilefilteredData.getRequestType();
			if(!filteredList.isEmpty()) {
				appendfilteredList.addAll(filteredList);
			}else {
				appendfilteredList.addAll(activityList);
			}
			filteredList.clear();
			for(activityDTO activityDto : appendfilteredList) {
				for(String requestType : listOfRequestType) {
					if(activityDto.getItemType().equalsIgnoreCase(requestType) && !(filteredList.contains(activityDto))) {
						filteredList.add(activityDto);
					}
			}
			}
			appendfilteredList.clear();
		}
		if(!(CollectionUtils.isNullOrEmpty(profilefilteredData.getAssignedTo()))) {
			listOfAssignedTo = profilefilteredData.getAssignedTo();
			if(!filteredList.isEmpty()) {
				appendfilteredList.addAll(filteredList);
			}else {
				appendfilteredList.addAll(activityList);
			}
			filteredList.clear();
			for(activityDTO activityDto : appendfilteredList) {
				for(String assignedTo : listOfAssignedTo) {
					if(activityDto.getAssignedUser().equalsIgnoreCase(assignedTo) && !(filteredList.contains(activityDto))) {
						filteredList.add(activityDto);
					}
			}
			}
			appendfilteredList.clear();
		}
		if(!(CollectionUtils.isNullOrEmpty(profilefilteredData.getAging()))) {
			int greaterThan90 = 0, startRange = 0, endRange = 0;
			listOfAging = profilefilteredData.getAging();
			if(!filteredList.isEmpty()) {
				appendfilteredList.addAll(filteredList);
			}else {
				appendfilteredList.addAll(activityList);
			}
			filteredList.clear();
			for(activityDTO activityDto : appendfilteredList) {
				for(String aging : listOfAging) {
					List<String> listOfNumbers = removeSpecialCharactersFromAging(aging);
					if(listOfNumbers.size()==1) {
						greaterThan90 = Integer.valueOf(listOfNumbers.get(0));
					}else {
						startRange = Integer.valueOf(listOfNumbers.get(0));
						endRange = Integer.valueOf(listOfNumbers.get(1));
					}
					int dateDifference = commonUtils.calculateDateDifference(activityDto.getCreatedDate(), false);
					if(greaterThan90!=0 && dateDifference>=greaterThan90 && !(filteredList.contains(activityDto))) {
						filteredList.add(activityDto);
						break;
					}
					else if(dateDifference>=startRange && dateDifference<=endRange && !(filteredList.contains(activityDto))){
						filteredList.add(activityDto);
						break;
					}
			}
			}
			appendfilteredList.clear();
		}
		if(profilefilteredData.isAwaitingFeedback()) {
			if(!filteredList.isEmpty()) {
				appendfilteredList.addAll(filteredList);
			}else {
				appendfilteredList.addAll(activityList);
			}
			filteredList.clear();
			for(activityDTO activityDto : appendfilteredList) {
					if(activityDto.getFeedBackUserId()!=null && activityDto.getFeedBackUserName()!=null && activityDto.getFeedBackUserEmail()!=null && !(filteredList.contains(activityDto))) {
						filteredList.add(activityDto);
					}
			}
		}
		
		if(!filteredList.isEmpty()) {
			return filteredList;
		} else {
			return activityList;
		}
	}
	
	private List<String> removeSpecialCharactersFromAging(String aging) {
		List<String> listOfAgingNumbers = new LinkedList<String>();
		
		if(aging.contains("-")) {
			String[] str= aging.split("-");
			listOfAgingNumbers.add(str[0].trim());
			listOfAgingNumbers.add(str[1].trim());
		} 
		if(aging.contains(">")) {
			aging = aging.replace(">", "");
			aging.trim();
			listOfAgingNumbers.add(aging);
		}
		return listOfAgingNumbers;
	}

	@Override

		activieRecordsByAgeFilteredData.setReportType(filterDTO.getReportType());
		activieRecordsByAgeFilteredData.setUserId(filterDTO.getUserId().get(0));
		return activieRecordsByAgeFilteredData;
	}
